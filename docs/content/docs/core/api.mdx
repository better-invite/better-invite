# Create Invite

<APIMethod path="/invite/create" method="POST" requireSession note="All parameters default to the plugin's options defaults, except for role and email.">
```ts
type createInvite = {
	/**
	 * The role to give the invited user.
	 */
	role: string

	/**
	 * The email (or emails) address of the user to send an invitation email to.
	 * If email is present, the invite is private.
	 */
	email?: string | string[]

	/**
	 * Type of token to use:
	 * 24 character token, 6 digit code or custom options.generateToken.
	 */
	tokenType?: "token" | "code" | "custom"

	/**
	 * The URL to redirect the user to create their account.
	 * {callbackUrl} will be replaced by the actual callback URL.
	 */
	redirectToSignUp?: string

	/**
	 * The URL to redirect the user to upgrade their role.
	 * {callbackUrl} will be replaced by the actual callback URL.
	 */
	redirectToSignIn?: string

	/**
	 * The number of times an invitation can be used.
	 * If not defined and maxUsesPerUser is defined, maxUses will be infinite.
	 */
	maxUses?: number
	/**
	 * The number of times an invitation can be used by the same user.
	 * Use `Infinity` for unlimited uses per user.
	 * Only works for private invites, public invites are always unlimited per user.
	 */
	maxUsesPerUser?: number

	/**
	 * Number of seconds the invitation token is valid for.
	 */
	expiresIn?: number

	/**
	 * The URL to redirect the user to after upgrading their role
	 * (after accepting the invite).
	 * {token} will be replaced with the actual invite token.
	 */
	redirectToAfterUpgrade?: string

	/**
	 * Whether the inviter's name should be shared with the invitee.
	 */
	shareInviterName?: boolean

	/**
	 * How should the sender receive the token.
	 * Sender only receives a token if no email is provided.
	 */
	senderResponse?: "token" | "url"

	/**
	 * Where should we redirect the user?
	 * Only if no email is provided.
	 */
	senderResponseRedirect?: "signUp" | "signIn"

	/**
	 * The URL where the user will be redirected to accept their invite.
	 * Use {token}, {signInUpUrl}, {callbackUrl}, {email} and {defaultUrlQuery}, which will be replaced with their values.
	 */
	customInviteUrl?: string
}
```
</APIMethod>

# Accept Invite

<APIMethod path="/invite/accept" method="POST" requireSession note="If the user is not logged in, it returns a message to sign in/up and stores the token in a cookie.">
```ts
type acceptInvite = {
	/**
	 * Where to redirect the user after accepting the invite.
	 * {token} will be replaced by the actual token in the request body.
	 *
	 * @default http://localhost:3000/
	*/
	callbackUrl?: string
	/**
	 * Where to redirect the user to sign in/up.
	 * {callbackUrl} will be replaced by the actual callbackUrl in the request body.
	*/
	signInUpUrl?: string
  /**
   * The invite token
  */
  token: string
	/**
	 * The email address of the user to sign in/up to.
	*/
	email?: string
}
```
</APIMethod>

# Accept Invite Callback

<Callout type="info" title="Note">
  This endpoint is **not recommended** for direct usage. It exists mainly for internal use (e.g. the default link sent in invite emails).
  The email link opens in the browser and triggers a `GET` request to `/api/auth/invite/{token}?callbackUrl=...`.
  For most apps, you should use [Accept Invite](#accept-invite) instead.
</Callout>

<APIMethod path="/invite/:token" clientMethod='invite[":token"]' method="GET" optionalSession fetchOptionsParams="token" note="If the user is not logged in, it redirects the user to the callback URL with the token in a cookie.">
```ts
type acceptInviteCallback = {
	/**
	 * Where to redirect the user after sing in/up
	 * {token} will be replaced by the actual token in the request body.
	 *
	 * @default http://localhost:3000/
	*/
	callbackUrl?: string
	/**
	 * Where to redirect the user to sign in/up.
	 * {callbackUrl} will be replaced by the actual callbackUrl in the request body.
	*/
	signInUpUrl?: string
	/**
   * @ignore
   * The invite token
  */
  token: string
	/**
	 * The email address of the user to sign in/up to.
	*/
	email?: string
}
```
</APIMethod>

# Get Invite

Get basic information about an invitation.

<Callout type="info" title="Note">
  This endpoint no longer returns `newAccount`.

  If you need to know whether an invited email will create a new account or upgrade an existing one, you can check whether a user already exists for that email:

  ```ts
  const invitedUser = await ctx.context.internalAdapter.findUserByEmail(
    email,
  );
  ```

  If `invitedUser` is `undefined`, the user does not exist yet, so the invitation will create a new account.
  Otherwise, the invitation will upgrade the role of an existing account.
</Callout>

<APIMethod path="/invite/get" method="GET" optionalSession note="For private invites, the requester must match the invite email.">
```ts
type getInvite = {
  /**
   * The invite token to look up.
	 * If not provided, the endpoint will search for the invite token in the invite cookie.
  */
  token?: string
}
```
</APIMethod>

# List Invites

List all (private) invitations for the current user

<APIMethod path="/invite/list" method="GET" requireSession note="Only lists PRIVATE invitations for the current user.">
```ts
type listInvites = {
	/**
	 * The value to search for
	 */
	searchValue?: string
	/**
	 * The field to search in, defaults to email. Can be `email`, `name` or `domainWhitelist`
	 */
	searchField?: "name" | "email" | "domainWhitelist"
	/**
	 * The operator to use for the search. Can be `contains`, `starts_with` or `ends_with`
	 */
	searchOperator?: "contains" | "starts_with" | "ends_with"
	/**
	 * The numbers of invitations to return
	 */
	limit?: number
	/**
	 * The offset to start from
	 */
	offset?: number
	/**
	 * The field to sort by
	 */
	sortBy?: string
	/**
	 * The direction to sort by. Can be `asc` or `desc`
	 */
	sortDirection?: "asc" | "desc"
	/**
	 * The field to filter by
	 */
	filterField?: string
	/**
	 * The value to filter by
	 */
	filterValue?: string | number | boolean
	/**
	 * The operator to use for the filter. Can be `eq`, `ne`, `lt`, `lte`, `gt` or `gte`
	 */
	filterOperator?: "eq" | "ne" | "lt" | "lte" | "gt" | "gte"
}
```
</APIMethod>
